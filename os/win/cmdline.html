Start - End ScriptÂ¶

skeleton.batÂ¶
:start
  @echo off
  setlocal
  set cmd_location="%~dp0"
  pushd %cmd_location%
  set SEPARATOR="--------------------------------------------------------------------------------"
  set INDENT="  "

  echo %SEPARATOR%
  echo "-- %~nx0 Started!"
  echo.

:script

:end
  echo.
  echo "-- %~nx0 Finished!"
  echo %SEPARATOR%
  popd
  endlocal
  goto:eof

Commandline ArgumentsÂ¶

cmnd_args.batÂ¶
::------------------------------------------------------------------------------
:: Parse command line options
::
:parse
set usage1="Usage: hdl_designer.bat [-v] [-h]"
set usage2="                        [-n designName]"
set usage3="                        [-d designDirectory]"

echo "Search Commandline Parameters"
:parseloop
if not "%1"=="" (
    if "%1"=="-v" (
        set VERBOSE=1
        echo "%INDENT:"=%verbose enabled"
        shift
    )
    if "%1"=="-h" (
        goto :HELP
        shift
    )
    if "%1"=="-n" (
        set design_name=%2
        echo "%INDENT:"=%design_name=!design_name:"=! "
        shift & shift
    )
    if "%1"=="-d" (
        set design_directory=%2
        echo "%INDENT:"=%design_directory=!design_directory:"=! "
        shift & shift
    )
    goto :parseloop
)
echo.

::------------------------------------------------------------------------------
:: Helper Functions
::
:HELP
  echo.
  echo %usage1:"=%
  echo %usage2:"=%
  echo %usage3:"=%
  echo.&pause&goto:eof
